## python示例代码

```python
#!/usr/bin/python
#coding: utf-8

import requests
import json
import ssl

# 协议头
busi_head_ = {
    "authorization_" : "your authorization_",
    "account_" : "59A9A1AA9DCAED242F711A4FE10FC51B",
    "account_type_" : 1,
    "game_id_"  : 19115,
    "world_id_" : 3012,
    "cmd_" : 1
}

# 角色登入详细数据
login = [  
    19115,                  # 1     game_id_
    "2019-12-24 11:02:03",  # 2     dteventtime 事件发生的时间
    1,                      # 3     platid ios=0/android=1
    1,                      # 4     account_type_ 帐号类型
    3012,                   # 5     world_id_  小区号
    "59A9A1AA9DCAED242F711A4FE10FC51B",   # 6     account_
    "987654321",            # 7     roleid 角色id 
    "龙魂乄圣主",             # 8     rolename 角色名，注意过滤特殊字符
    3,                      # 9     job 职业
    45,                     # 10    rolelevel 等级
    123456,                 # 11    fightpower 战力
    8,                      # 12    friendsnum 好友数量
    "WIFI",                 # 13    network  WIFI/5G/4G/3G/2G
    "110.89.251.140",       # 14    clientip 客户端ip地址,   注意不是局域网ip地址
    "866205043997954",      # 15    deviceid 设备ID,ios可取idfv,android可取imei
    "2019-01-02 03:04:05",  # 16    dtfirstcreate    首次注册时间 
    8640000                 # 17    totalonlinetime  累计在线时长(秒)
]
    

data_json = {
    "busi_head_" : busi_head_,
    "busi_body_" : {
        # 详细日志用竖线分隔, 拼接成字符串
        "text_":  '|'.join( [str(x) for x in login] )
    }
}

# debug
print json.dumps(data_json, indent=2)

# 上传数据
header = {'Content-type' : 'application/json;charset=utf-8'}
ssl._create_default_https_context = ssl._create_unverified_context
req = requests.post('https://services.anticheatexpert.com/studio', headers=header, data=json.dumps(data_json))

# 返回结果
print "response:", req.text
```
